# 三、流式实现（后端必须照做）

---

## Node示例（最小实现）

```js
app.post('/api/chat/stream', async (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  const { session_id, character_id, message } = req.body;

  const session = getOrCreateSession(session_id);
  const character = getCharacter(character_id);

  if (!character) {
    res.write(`data: ${JSON.stringify({ type: "error", message: "INVALID_CHARACTER" })}\n\n`);
    return res.end();
  }

  session.messages.push({ role: "user", content: message });

  const systemPrompt = buildSystemPrompt(character);
  const recentMessages = session.messages.slice(-10);

  const finalMessages = [
    { role: "system", content: systemPrompt },
    ...recentMessages
  ];

  res.write(`data: ${JSON.stringify({ type: "start" })}\n\n`);

  const stream = await callLLMStream(finalMessages);

  let fullText = "";

  for await (const chunk of stream) {
    const token = chunk.choices[0]?.delta?.content || "";
    if (!token) continue;

    fullText += token;

    res.write(`data: ${JSON.stringify({
      type: "delta",
      content: token
    })}\n\n`);
  }

  session.messages.push({
    role: "assistant",
    content: fullText
  });

  res.write(`data: ${JSON.stringify({ type: "end" })}\n\n`);
  res.end();
});
```

# 四、前端对接（你重点）

---

## 小程序实现方式（核心）

👉 小程序不原生支持 EventSource，需要：

### 方案：`wx.request + chunked`

---

## 核心逻辑（伪代码）

```js
wx.request({
  url: '/api/chat/stream',
  method: 'POST',
  enableChunked: true, // 关键
  data: {...},

  success(res) {},

  onChunkReceived(chunk) {
    const text = decode(chunk.data);

    const events = text.split("\n\n");

    events.forEach(e => {
      if (!e.startsWith("data:")) return;

      const json = JSON.parse(e.replace("data: ", ""));

      if (json.type === "delta") {
        appendToUI(json.content);
      }
    });
  }
});
```

---

## UI处理策略（必须这样做）

```text
delta → 直接拼接到当前文本
end → 标记完成
```
